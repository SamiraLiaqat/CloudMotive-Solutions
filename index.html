<!-- JavaScript for D3 Chart and Icon Initialization -->

<script>

// Initialize Lucide Icons

lucide.createIcons();

 

// Data for the graph

const successData = [

{ metric: "Project Success Rate", value: 98, target: 90 },

{ metric: "Client Retention (YOY)", value: 95, target: 85 },

];

 

function createBarChart() {

// Select the container and get its actual width for responsiveness

const container = document.getElementById('success-chart');

if (!container) return; // Exit if container not found

 

// Standard margins and calculated dimensions

const margin = { top: 20, right: 30, bottom: 40, left: 130 };

const containerWidth = container.clientWidth;

const containerHeight = 300;

const width = containerWidth - margin.left - margin.right;

const height = containerHeight - margin.top - margin.bottom;

 

// Append SVG element to the container

const svg = d3.select("#success-chart")

.append("svg")

.attr("width", containerWidth)

.attr("height", containerHeight)

.append("g")

.attr("transform", `translate(${margin.left},${margin.top})`);

 

// X scale (Value, from 0 to 100%)

const x = d3.scaleLinear()

.domain([0, 100])

.range([0, width]);

 

// Y scale (Metric labels, Band scale for bars)

const y = d3.scaleBand()

.domain(successData.map(d => d.metric))

.range([0, height])

.padding(0.4); // Increased padding for separation

 

// Draw Y axis labels

svg.append("g")

.call(d3.axisLeft(y).tickSize(0).tickPadding(10))

.selectAll("text")

.attr("fill", "#94A3B8") // Gray text

.style("font-size", "14px")

.style("font-weight", "600");

 

// Draw X axis

svg.append("g")

.attr("transform", `translate(0,${height})`)

.call(d3.axisBottom(x).ticks(5).tickFormat(d => d + "%"))

.selectAll("text")

.attr("fill", "#94A3B8")

.style("font-size", "12px");

 

// Background Bars (Industry Target)

svg.selectAll(".target-bar")

.data(successData)

.enter().append("rect")

.attr("class", "target-bar")

.attr("x", 0)

.attr("y", d => y(d.metric))

.attr("height", y.bandwidth())

.attr("width", d => x(d.target))

.attr("fill", "#334155") // Slate-700 dark gray background

.attr("rx", 6)

.attr("ry", 6);

 

// Actual Success Bars (Animated)

svg.selectAll(".success-bar")

.data(successData)

.enter().append("rect")

.attr("class", "success-bar")

.attr("x", 0)

.attr("y", d => y(d.metric))

.attr("height", y.bandwidth())

.attr("fill", "#00E0FF") // Cyan fill

.attr("rx", 6)

.attr("ry", 6)

.attr("width", 0) // Start at zero width for animation

.transition()

.duration(1500)

.ease(d3.easeQuadOut)

.attr("width", d => x(d.value));

 

// Value Labels (Text overlay)

svg.selectAll(".value-label")

.data(successData)

.enter().append("text")

.attr("class", "value-label")

.attr("x", d => x(d.value) + 12)

.attr("y", d => y(d.metric) + y.bandwidth() / 2 + 5)

.text(d => d.value + "%")

.attr("fill", "white")

.style("font-weight", "bold")

.style("font-size", "16px");

 

// Target Labels (Text for industry average)

svg.selectAll(".target-label")

.data(successData)

.enter().append("text")

.attr("class", "target-label")

.attr("x", d => x(d.target) - 5) // Position just inside the target bar

.attr("y", d => y(d.metric) + y.bandwidth() / 2 + 5)

.text(d => d.target + "%")

.attr("fill", "#94A3B8")

.style("font-weight", "600")

.style("font-size", "14px")

.style("text-anchor", "end");

}

 

// Run functions after window loads

window.onload = function() {

createBarChart();

 

// Rerun the chart on window resize for responsiveness

window.addEventListener('resize', () => {

d3.select("#success-chart").select("svg").remove(); // Remove old chart

createBarChart(); // Draw new chart

});

};

</script>
